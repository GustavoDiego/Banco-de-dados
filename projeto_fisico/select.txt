SELECT nome, valor_ingresso 
FROM Evento 
WHERE valor_ingresso > 0 
ORDER BY valor_ingresso DESC;

SELECT id_local, COUNT(*) AS TotalEventos
FROM Evento
GROUP BY id_local;

SELECT id_local, AVG(valor_ingresso) AS MediaValor
FROM Evento
GROUP BY id_local
HAVING AVG(valor_ingresso) > 50;

SELECT P.nome, E.nome AS Evento
FROM Participante P
JOIN Participacao PA ON P.id = PA.id
JOIN Evento E ON PA.id_evento = E.id;

SELECT nome
FROM Local
WHERE id IN (
    SELECT id_local 
    FROM Evento 
    WHERE valor_ingresso > 0);
